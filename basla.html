/* 
* İlk C++ programımızda merhaba diyelim
 */
#include <iostream>     // Giriş/Çıkış işlemleri için bu kütüphaneye ihtiyacımız var
using namespace std;
 
int main() {                        // Program giriş yeri
   cout << "Merhaba, Dünya" << endl;  // Merhaba 
   return 0;                       
}                                   // Main fonksiyonunun bitişi
Programın Kısa Açıklaması
  1. Çok Satırlı Yorum: /* ile başlar ve */ ile biter. Birden fazla satıra yayılabilir (1-3 satırlarında olduğu gibi).
  2. Satır Sonu Yorumu: // ile başlar ve geçerli satırın sonuna kadar sürer. (4, 7, 8, 9 ve 10 satırlarında olduğu gibi).

#include <iostream>
using namespace std;
 " #include" dahil etmek anlamında kullanılır.Derleme öncesi işlenir. "#include <iostream>" ifadesi, önişlemciye giriş/çıkış işlemlerini desteklemek için "iostream" kütüphanesini içermesini söyler."using namespace std;" Türkçe karşığı isim uzayı olan namespace kavramı, kullanıcı tarafından oluşturulan veya öntanımlı olarak gelen kütüphaneleri mantıksal bir hiyararşi içerisinde tutmak için kullanılır. Basit olarak C++ dilinde namespace kullanımına en çok kullanılan cout nesnesi üzerinden bakalım. Ekrana çıktı vermeye yarayan ostream sınıfından oluşan cout nesnesi mantıksal olarak std namespace içerisinde yer alır.Eğer programımız boyunca cout nesnesi çokça kullanılacaksa namespace tanımlaması dosyanın en başında yapılıp, tüm program boyunca bütün fonksiyonlardan erişilmesine olanak sağlanmış olur.

int main()
main() programın başlangıç noktası olan ana fonksiyondur. main() bir int(tam sayı) döndürmesi gerekir.

cout << "Merhaba, Dünya" << endl;
" cout"standart çıktıya (veya Konsol ÇIKIŞI'na) işaret eder.<< sembolü "hello, world" yazısını konsola koymak için kullanılan akış ekleme operatörüdür. "endl" imleci sonraki satıra getirmek için kullanılan operatördür.

return 0;
main() fonksiyonunun hatasız sonlandırıldığı anlamına gelir.

"cout <<" üzerinden çıktı

C ++'da ekran konsoluna çıkış "cout" ve akış ekleme operatörü aracılığıyla yapılır.<< size istediğiniz kadar çok sayıda öğe yazdırma imkanı sağlar.Örneğin; 

    cout << "merhaba" << " dünya, " <<  "tekrar!" << endl;
    cout << "merhaba," << endl << "bir kez daha. " << endl;

Çıktı:

    merhaba dünya, tekrar!
    merhaba,
    bir kez daha.

İmleci bir sonraki satıra ilerletmek için yeni satır karakterini belirten endl ifadesini kullanacağımızı öğrenmiştik. Benzer şekilde bir sonraki satıra ilerlemek veya imleci bir tab kadar atlatmak için bazı karakterler kullanabiliriz. Bunlar; '\n' ve '\t' 

    cout << "merhaba dünya, tekrar!\n";
    cout << "\t merhaba,\n bir\t kez\t daha.\n";

Kodunuzun çıktısı şöyle görünecektir:

        merhaba dünya, tekrar!
        merhaba,
        bir     kez    daha.

/*
 * Kullanıcıdan istenen iki sayının toplamını,farkını,çarpımını ve bölümünü yazdır
 * (DortIslem.cpp)
 */
#include <iostream> 
using namespace std;
 
int main() {
   int ilkSayi;  // ilkSayi adında int tipinde bir değişken
   int ikinciSayi; // ikinciSayi adında int tipinde bir değişken
   int toplam; fark, carpim, bolum;
                  // Sonuçların saklanması için int tipinde 4 değişken
 
   cout << "İlk Sayiyi Giriniz: ";   // Giriş istenmesi
   cin >> ilkSayi;                   // Girilen sayıyı ilkSayi değişkenine atmak
   cout << "İkinci Sayiyi Giriniz: ";  // Giriş istenmesi
   cin >> ikinciSayi;                  // Girilen sayıyı ikinciSayi değişkenine atmak
 
   // İşlemler gerçekleştiriliyor
   toplam    = ilkSayi + ikinciSayi;
   fark      = ilkSayi - ikinciSayi;
   carpim    = ilkSayi * ikinciSayi;
   bolum     = ilkSayi / ikinciSayi;
 
   // Sonuçları Yazdır
   cout << "Toplam: " << toplam << endl;
   cout << "Fark: " << fark << endl;
   cout << "Çarpım: " << carpim << endl;
   cout << "Bölüm: " << bolum << endl;
 
   return 0;
}

Bir "cin" ifadesinde birden fazla değer okuma

cout << "İki Sayi Giriniz:  (boşluklarla ayrılmış olarak): ";  // Bir uyarı mesajı koyduk
cin >> ilkSayi >> ikinciSayi;                        // İki değeri bir cin ifadesinde okuduk
